# Find all minimal sets and return union of all premises
def find_all_minimal_premises(premises_dict, conclusion):
    """
    Find all minimal valid subsets, then return the union of all premises
    that appear in at least one minimal set.
    """
    minimal_sets = []
    premise_keys = list(premises_dict.keys())
    n = len(premise_keys)
    
    # Reuse solver for efficiency
    solver = Solver()

    for subset_size in range(1, n + 1):
